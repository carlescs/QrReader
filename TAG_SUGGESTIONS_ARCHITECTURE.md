## State Management

```kotlin
// ViewModel state example

data class QrCameraUiState(
    // Existing state
    val scannedBarcode: String? = null,
    val isScanning: Boolean = true,
    
    // New tag suggestion state
    val tagSuggestions: List<String> = emptyList(),
    val isLoadingSuggestions: Boolean = false,
    val suggestionsError: String? = null,
    val selectedTags: Set<String> = emptySet(),
    
    // Feature availability
    val geminiNanoStatus: GeminiNanoStatus = GeminiNanoStatus.UNAVAILABLE
)

// ViewModel implementation
class QrCameraViewModel(
    private val generateTagSuggestions: GenerateTagSuggestionsUseCase,
    private val checkAvailability: CheckGeminiNanoAvailabilityUseCase
) : ViewModel() {
    
    private val _uiState = MutableStateFlow(QrCameraUiState())
    val uiState: StateFlow<QrCameraUiState> = _uiState.asStateFlow()
    
    init {
        checkGeminiNanoAvailability()
    }
    
    fun onBarcodeScanned(content: String, type: Int) {
        _uiState.update { it.copy(scannedBarcode = content) }
        
        if (uiState.value.geminiNanoStatus == GeminiNanoStatus.AVAILABLE) {
            loadTagSuggestions(content, type)
        }
    }
    
    private fun loadTagSuggestions(content: String, type: Int) {
        viewModelScope.launch {
            _uiState.update { it.copy(isLoadingSuggestions = true) }
            
            val result = generateTagSuggestions(content, type)
            
            _uiState.update { 
                it.copy(
                    isLoadingSuggestions = false,
                    tagSuggestions = result.getOrNull()?.suggestions ?: emptyList(),
                    suggestionsError = result.exceptionOrNull()?.message
                )
            }
        }
    }
}
```
